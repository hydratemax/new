app.get("/search", async (req, res) => { try { const q = (req.query.q || "").toLowerCase().trim(); const pages = ["/home", "/filter?keyword=" + encodeURIComponent(q)]; let results = []; for (const p of pages) { try { const r = await axios.get(BASE + p, { headers: HEADERS, timeout: 12000 }); const $ = cheerio.load(r.data); $(".unit").each((i, el) => { const a = $(el).find("a").first(); const href = a.attr("href"); if (!href || !href.includes("/manga/")) return; const title = $(el).find(".info a").first().text().trim() || $(el).find("img").attr("alt") || ""; if (!title) return; const id = href.replace("/manga/", "").replace("/", ""); const img = $(el).find("img").attr("src") || $(el).find("img").attr("data-src") || ""; // fuzzy match (IMPORTANT FIX) const cleanTitle = title.toLowerCase(); if (q && !cleanTitle.includes(q) && !q.includes(cleanTitle)) return; results.push({ id, title, cover: img }); }); } catch (e) { console.log("PAGE FAIL:", p, e.message); } } // remove duplicates results = results.filter((v, i, a) => a.findIndex(t => t.id === v.id) === i ); return res.json({ results }); } catch (e) { console.log("SEARCH ERROR:", e.message); return res.json({ results: [] }); } });
