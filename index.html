app.get("/search", async (req, res) => { try { const q = (req.query.q || "").toLowerCase(); const r = await axios.get(`${BASE}/home`, { headers: HEADERS, timeout: 12000 }); const $ = cheerio.load(r.data); const results = []; $(".unit, .swiper-slide").each((i, el) => { const a = $(el).find("a").first(); const href = a.attr("href") || ""; const title = $(el).find(".info a, .title, a").first().text().trim(); const img = $(el).find("img").attr("src") || $(el).find("img").attr("data-src") || ""; if (!href.includes("/manga/") || !title) return; // keyword match fallback if (!title.toLowerCase().includes(q)) return; const id = href.replace("/manga/", "").replace("/", ""); results.push({ id, title, cover: img }); }); return res.json({ results }); } catch (e) { console.log("SEARCH ERROR:", e.message); return res.json({ results: [] }); } });
